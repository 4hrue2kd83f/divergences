dvg_annex_equinoxea = {
	is_shown = {
		exists = c:EQX
		root = {
			has_state_in_state_region = STATE_FRANCE_ANTARTIQUE
		}
		NOT = {
			has_variable = dvg_equinoxea_wars
		   
		}
		NOT = { has_global_variable = equinoxea_annex }
	}

    possible = {
		
		NOT = {
		 has_truce_with = c:EQX 
		 }
	}

    when_taken = {
		c:EQX = { trigger_event = { id = dvg_amazonie.5 popup = yes } }
		set_global_variable = equinoxea_annex
    }

	ai_chance = {
		base = 100
	}
}

dvg_annex_bresil = {
	is_shown = {
		exists = c:BRS
		root = {
			has_state_in_state_region = STATE_FRANCE_ANTARTIQUE
		}
		NOT = {
			has_variable = bresil_annex
		   
		}
	}

    possible = {
		
		NOT = {
		 has_truce_with = c:BRS 
		 }
		 any_neighbouring_state = {
				owner = c:BRS
			}
	}

    when_taken = {
		create_diplomatic_play = {
				
			
				target_country = c:BRS
				
				war = no
				
				type = dp_annex_war
				
				add_war_goal = {
					holder = c:BRS
					type = war_reparations
					target_country = c:AMA
				}
				
			}
    }
	set_global_variable = bresil_annex

	ai_chance = {
		base = 100
	}
}

dvg_seize_para = {
	is_shown = {
		
		country_has_primary_culture = cu:amazonien	
		
		
		NOT = {
			owns_entire_state_region = STATE_PARA
		}
		
	}

    possible = {
		
		has_state_in_state_region = STATE_CAPPADOCE	
		c:GRA = {
				is_at_war = yes
				relations:root < 0
			}
		
	}

    when_taken = {
		s:STATE_PARA.region_state:GRA = {
			set_state_owner = ROOT
		}
		c:GRA = { trigger_event = { id = dvg_amazonie.10 popup = yes } }
		change_relations = {
				country = c:SPA
				value = -30
			}
		change_relations = {
				country = c:GRA
				value = -30
			}
		create_incident = {
			country = c:GRA
			country = c:AMA
			value = 6
		}
    }

	ai_chance = {
		base = 100
	}
}
