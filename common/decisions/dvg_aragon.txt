dvg_ari_acceptitalians = {
	is_shown = {
		exists = c:ARN
		root = c:ARN
		
	}

	possible = {
		has_technology_researched = pan-nationalism
		owns_entire_state_region = STATE_CAMPANIA
		owns_entire_state_region = STATE_LAZIO
		owns_entire_state_region = STATE_CATALONIA
		owns_entire_state_region = STATE_CALABRIA
		OR = {
			owns_entire_state_region = STATE_SICILY
			owns_entire_state_region = STATE_SARDINIA
		}
	}
	
    when_taken = {
		change_tag = ARI
	
		add_primary_culture = cu:south_italian
		
    }

	ai_chance = {
		base = 100
	}
}

dvg_civilizer = {
	is_shown = {
		is_ai = yes
		any_neighbouring_state = {
			owner = {
				is_ai = yes
				is_country_type = unrecognized
			}
		}
	}

	possible = {
		has_technology_researched = civilizing_mission
		any_neighbouring_state = {
			owner = {
				NOT = {
					has_truce_with = ROOT
				}
				is_country_type = unrecognized
				

			}
		}
		is_diplomatic_play_committed_participant = no
	}
	
    when_taken = {
		random_neighbouring_state = {
			limit = {
				owner = {
					is_country_type = unrecognized
					NOT = {
						has_truce_with = ROOT
					} 
					relations:root <= relations_threshold:cold
				}
			}
			owner = {
				save_scope_as = civilizer_target
			}
		}
		create_diplomatic_play = {
			type = dp_annex_war
			target_country = scope:civilizer_target

			add_war_goal = {
				holder = scope:civilizer_target
				type = independence
			}
		}
    }

	ai_chance = {
		base = 100
	}
}
