dvg_afroarcadian_liberation = {
	is_shown = {
		
		exists = c:LIB
		c:LIB = ROOT

		NOT = {
			has_global_variable = afroarcadian_liberation
		}
	}

    possible = {
		has_technology_researched = pan-nationalism
		
		NOT = {
			ruler = {
				has_ideology = ideology:ideology_pacifist
			}
		}
		has_law = law_type:law_racial_segregation

	}
		
		

    when_taken = {
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:afro_arcadian
				is_homeland = cu:afro_latino
				is_homeland = cu:afro_germanic
				}
			state_is_in_north_america = yes
			}
			add_claim = ROOT
		}
		
		
		add_primary_culture = cu:afro_latino
		
		create_incident = {
			country = ROOT
			value = 20
		}
		set_global_variable = {
				name = afroarcadian_liberation
				value = 0
			}
    }

	ai_chance = {
		base = 100
	}
}

dvg_spread_homeland_cadien = {
	is_shown = {
		
		country_has_primary_culture = cu:cadien

		any_scope_state = {
			culture_percent_state = {
				culture = cu:cadien
				value >= 0.33
			}
			state_region = {
				NOT = {
					is_homeland = cu:cadien
				}
			}
		}
	}

    possible = {
		has_technology_researched = nationalism
		has_technology_researched = romanticism
	}
		
		

    when_taken = {
		every_scope_state = {
			limit = {
				culture_percent_state = {
					culture = cu:cadien
					value >= 0.33
				}
				state_region = {
					NOT = {
						is_homeland = cu:cadien
					}
				}
				state_is_in_north_america = yes
			}
			state_region = {
				add_homeland = cu:cadien
			}
		}
		
	
		
		
    }

	ai_chance = {
		base = 100
	}
}
dvg_manifest_destiny_cadien = {
	is_shown = {
		
		country_has_primary_culture = cu:cadien

		NOT = {
			has_variable = manifested_destiny_plantagenia
		}
	}

    possible = {
		has_technology_researched = nationalism
		has_technology_researched = romanticism
		country_rank >= rank_value:minor_power
		is_subject = no
		is_at_war = no 
		NOT = {
			has_law = law_type:law_no_colonial_affairs
		}
	}
    when_taken = {
		every_state = {
			limit = {
				
				state_region = {
					
				is_homeland = cu:cadien
					
				}
				NOT = {
					owner = ROOT
				}
				state_is_in_north_america = yes
			}
			state_region = {
					
				add_claim = ROOT
					
			}
			
			
		}
		s:STATE_BAHIA = {
			add_claim = ROOT
		}
		s:STATE_APACHERIA = {
			add_claim = ROOT
		}
		s:STATE_COMANCHERIE = {
			add_claim = ROOT
		}
		s:STATE_TERRAS_VERMELHAS = {
			add_claim = ROOT
		}
		s:STATE_TEJAS = {
			add_claim = ROOT
		}
		if = {
			limit = {
				OR = {
					ruler = {
						has_ideology = ideology:ideology_jingoist_leader
					}
					any_interest_group = {
						OR = {
							has_ideology = ideology:ideology_jingoist
							leader = {
								has_ideology = ideology:ideology_jingoist_leader
							}
						}
						is_powerful = yes
					}

				}
				
			}
			s:STATE_BAJA_CALIFORNIA = {
				add_claim = ROOT
			}
			s:STATE_XINGTIAN = {
				add_claim = ROOT
			}
		}
		set_variable = manifested_destiny_plantagenia
    }

	ai_chance = {
		base = 100
	}
}
