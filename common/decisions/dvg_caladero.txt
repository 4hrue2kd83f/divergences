	}
	possible = {
		is_at_war = no
	}
	when_taken = {
		if = {
			limit = { exists = s:STATE_NEAR_COLLASUYU.region_state:BEO }
			trigger_event = { id = dvg_caladero.115 }
		}
		else_if = {
			limit = {  
				exists = s:STATE_NEAR_COLLASUYU.region_state:QUL
				NOT = { c:QUL = { is_subject_of = c:BRG } }
			}
			trigger_event = { id = dvg_caladero.115 }

			limit = {
