
dvg_annex_taiping = {
	is_shown = {
	
		country_has_state_religion = rel:taiping
		OR = {
		c:JIA = {is_player = no}
		NOT = {exists = c:JIA}
		
		}
		
	}

    possible = {

		any_country = {
			AND = {
			country_has_state_religion = rel:taiping
			country_has_primary_culture = cu:han
			}
			
		}
	
	}

    when_taken = {
		every_country = {
			if = {
				limit = {
					is_ai = yes			
					country_has_state_religion = rel:taiping
				}			
				root = { annex = prev }
			}
		}
		create_incident = {
			country = ROOT
			value = 4
		}
    }

	ai_chance = {
		base = 100
	}
}
dvg_xibei_uighurs = {
	is_shown = {
	
		country_has_state_religion = rel:sunni
		exists = c:XIB
		c:XIB = ROOT
		NOT = {
			country_has_primary_culture = cu:uighur
		}
	}

    possible = {

		owns_entire_state_region = STATE_TIANSHAN 
		
		has_law = law_type:law_cultural_exclusion
	}

    when_taken = {
		add_primary_culture = cu:uighur
    }

	ai_chance = {
		base = 100
	}
}
