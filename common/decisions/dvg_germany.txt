boh_german_confederation = {
	is_shown = {
		exists = c:GER
		root = c:GER
		was_formed_from = BOH
		NOT = { has_variable = boh_german_confederation }
	}

	possible = {
		exists = c:GER
		root = c:GER
	}
	
    when_taken = {
		trigger_event = { id = german_unification.10 }
		set_variable = boh_german_confederation
    }

	ai_chance = {
		base = 100
	}
}

dvg_ger_accept_balters = {
	is_shown = {
		OR = {
			AND = {
				exists = c:GER
				root = c:GER
			}
			AND = {
				exists = c:HLR
				root = c:HLR
			}
		}
		NOT = { has_variable = dvg_ger_balters_accepted }
	}

	possible = {
		OR = {
			AND = {
				exists = c:GER
				root = c:GER
			}
			AND = {
				exists = c:HLR
				root = c:HLR
			}
		}
	}

    when_taken = {
		IF = {
			limit = {
				root = c:GER
			}
			add_primary_culture = cu:balters
		}
		IF = {
			limit = {
				root = c:HLR
			}
			add_primary_culture = cu:balters
		}
		set_variable = dvg_ger_balters_accepted
    }

	ai_chance = {
		base = 100
	}
}
