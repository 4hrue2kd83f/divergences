dvg_colombian_manifest_destiny = {
	is_shown = {
		OR = {
			exists = c:GCO
			exists = c:LUS
			exists = c:MEX
		}
		is_subject = no
	}
	
	possible = {
		has_technology_researched = nationalism
	}
	
	when_taken = {
		s:STATE_XINGTIAN = {
			add_claim = ROOT
		}
		s:STATE_JINSHAN = {
			add_claim = ROOT
		}
		s:STATE_YICHENG = {
			add_claim = ROOT
		}
	}
	
	ai_chance = {
		base = 100
	}
}

dvg_nominate_mirabal = {
	is_shown = {
		exists = c:GCO
		is_subject = no
		has_variable = castro_empire
		NOT = { has_variable = has_mirabal }
	}
	when_taken = {
		create_character = {
			historical = yes
			first_name = Juan_Carlos
			last_name = Mirabal
			culture = caribeno
			birth_date = 1803.01.24
			ideology = ideology_authoritarian
			ig_leader = yes
			interest_group = ig_petty_bourgeoisie
			on_created = {
				set_variable = has_mirabal
			}
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_no_police
					has_law = law_type:law_local_police
				}
			}
			activate_law = law_type:law_dedicated_police
			else = {
				limit = { has_law = law_type:law_dedicated_police }
			}
			activate_law = law_type:law_militarized_police
		}
	}
}
