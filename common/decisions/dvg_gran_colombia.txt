dvg_nominate_mirabal = {
	is_shown = {
		exists = c:GCO
		this = c:GCO
		is_subject = no
		exists =  global_var:castro_empire
		NOT = { has_variable = has_mirabal }
	}
	when_taken = {
		create_character = {
			historical = yes
			first_name = Juan_Carlos
			last_name = Mirabal
			culture = caribeno
			birth_date = 1803.01.24
			ideology = ideology_authoritarian
			ig_leader = yes
			interest_group = ig_petty_bourgeoisie
			on_created = {
				set_variable = has_mirabal
			}
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_no_police
					has_law = law_type:law_local_police
				}
			}
			activate_law = law_type:law_dedicated_police
			else = {
				limit = { has_law = law_type:law_dedicated_police }
			}
			activate_law = law_type:law_militarized_police
		}
	}
}

		country_rank >= rank_value:great_power
		exists = global_var:castro_empire
		NOT = { exists = global_var:marquess_oaxaca_employed }
	}
	possible = {
		country_rank >= rank_value:great_power
		gold_reserves >= 100000
	}
	when_taken = {
		gold_reserves = -100000
		create_character = {
			historical = yes
			first_name = Henrique
			last_name = de_Albuquerque_Cortes
			culture = portuguese
			birth_date = 1803.01.24
			ideology = ideology_liberal
			interest_group = ig_landowner
			on_created = {
				set_as_interest_group_leader = yes
			}
		}
		change_relations = {
			limit = {
				country_rank = rank_value:great_power
				NOT = { country = c:SPA }
			}
			value = 100
		}
		change_relations = {
			country = c:PLA
			value = 50
		}
		set_global_variable = {
			name = marquess_oaxaca_employed
			value = 0
		}
	}
}
