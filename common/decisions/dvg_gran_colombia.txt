dvg_colombian_manifest_destiny = {
	is_shown = {
		OR = {
			exists = c:GCO
			exists = c:LUS
			exists = c:MEX
		}
		is_subject = no
	}
	
	possible = {
		has_technology_researched = nationalism
	}
	
	when_taken = {
		s:STATE_XINGTIAN = {
			add_claim = ROOT
		}
		s:STATE_JINSHAN = {
			add_claim = ROOT
		}
		s:STATE_YICHENG = {
			add_claim = ROOT
		}
	}
	
	ai_chance = {
		base = 100
	}
}
dvg_colombian_expansion = {
	is_shown = {
		OR = {
			exists = c:GCO
			exists = c:LUS
			exists = c:MEX
		}
		is_subject = no
	}
	
	possible = {
		owns_entire_state_region = STATE_XINGTIAN
		owns_entire_state_region = STATE_JINSHAN
		owns_entire_state_region = STATE_YICHENG
	}
	
	when_taken = {
		if = {
			limit = {
				country_has_primary_culture = cu:lusitanian
			}
			s:STATE_JINSHAN = {
				add_homeland = cu:lusitanian
				add_modifier = {
					name = dvg_open_door_policy
					years = 20
				}
			}
			s:STATE_YICHENG = {
				add_homeland = cu:lusitanian
				add_modifier = {
					name = dvg_open_door_policy
					years = 20
				}
			}
			else_if = {
				limit = {
					AND = {
						country_has_primary_culture = cu:lusitanian
						owns_entire_state_region = STATE_COMANCHERIE
					}
					s:STATE_XINGTIAN = {
						add_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
					s:STATE_JINSHAN = {
						add_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
					s:STATE_YICHENG = {
						add_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
					s:STATE_COMANCHERIE = {
						add_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
				}
			}
			else_if = {
				limit = {
					AND = {
						NOT = { country_has_primary_culture = cu:lusitanian }
						country_has_primary_culture = cu:mexican
					}
					s:STATE_XINGTIAN = {
						add_homeland = cu:mexican
						remove_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_mexicanization_lusitania
							years = 20
						}
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
					s:STATE_JINSHAN = {
						add_homeland = cu:mexican
						remove_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_mexicanization_lusitania
							years = 20
						}
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
					s:STATE_YICHENG = {
						add_homeland = cu:mexican
						remove_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_mexicanization_lusitania
							years = 20
						}
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
				}
			}
			else = {
				limit = {
					AND = {
						NOT = { country_has_primary_culture = cu:lusitanian }
						country_has_primary_culture = cu:mexican
						owns_entire_state_region = STATE_COMANCHERIE
					}
					s:STATE_XINGTIAN = {
						add_homeland = cu:mexican
						remove_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_mexicanization_lusitania
							years = 20
						}
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
					s:STATE_JINSHAN = {
						add_homeland = cu:mexican
						remove_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_mexicanization_lusitania
							years = 20
						}
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
					s:STATE_YICHENG = {
						add_homeland = cu:mexican
						remove_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_mexicanization_lusitania
							years = 20
						}
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
					s:STATE_COMANCHERIE = {
						add_homeland = cu:mexican
						remove_homeland = cu:lusitanian
						add_modifier = {
							name = dvg_mexicanization_lusitania
							years = 20
						}
						add_modifier = {
							name = dvg_open_door_policy
							years = 20
						}
					}
				}
			}
		}
	}
	
	ai_chance = {
		base = 100
	}
}
