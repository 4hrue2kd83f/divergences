dvg_colombian_empire = {
	is_shown = {
		AND = {
			exists = c:GCO
			this = c:GCO
			is_subject = no
			has_variable = castro_colombia
		}
		NOT = { exists = global_var:castro_empire }
	}
	possible = {
		country_rank = rank_value:great_power
		owns_entire_state_region = STATE_CUBA
	}
	when_taken = {
		set_global_variable = {
			name = castro_empire
			value = 0
		}
		trigger_event = { id = dvg_gran_colombia.24 }
	}
}

dvg_nominate_mirabal = {
	is_shown = {
		exists = c:GCO
		this = c:GCO
		is_subject = no
		exists = { global_var:castro_empire }
		NOT = { has_variable = has_mirabal }
	}
	when_taken = {
		create_character = {
			historical = yes
			first_name = Juan_Carlos
			last_name = Mirabal
			culture = caribeno
			birth_date = 1803.01.24
			ideology = ideology_authoritarian
			ig_leader = yes
			interest_group = ig_petty_bourgeoisie
			on_created = {
				set_variable = has_mirabal
			}
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_no_police
					has_law = law_type:law_local_police
				}
			}
			activate_law = law_type:law_dedicated_police
			else = {
				limit = { has_law = law_type:law_dedicated_police }
			}
			activate_law = law_type:law_militarized_police
		}
	}
}
dvg_marquess_oaxaca = {
	is_shown = {
		exists = c:GCO
		this = c:GCO
		exists = c:SPA
		is_subject = no
		has_law = law_type:law_monarchy
		exists = { global_var:castro_empire }
		NOT = { exists = global_var:marquess_oaxaca_employed }
	}
	possible = {
		exists = c:SPA
		c:SPA = {
			is_at_war = no
			has_law = law_type:law_monarchy
			is_ai = yes
			relations:root >= relations_threshold:amicable
		}
	}
	when_taken = {
		create_character = {
			historical = yes
			first_name = Henrique
			last_name = de_Albuquerque_Cortes
			culture = portuguese
			birth_date = 1803.01.24
			ideology = ideology_liberal
			interest_group = ig_landowner
			on_created = {
				set_as_interest_group_leader = yes
			}
		}
		change_relations = {
			country = c:SPA
			value = 100
		}
		set_global_variable = {
			name = marquess_oaxaca_employed
			value = 0
		}
	}
}
