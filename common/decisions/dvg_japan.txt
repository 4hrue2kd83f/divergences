dvg_sandakan = {
	is_shown = {
		exists = c:JAP
		exists = c:SDK
		OR = {
			c:SDK = ROOT
			c:JAP = ROOT
		}
		is_player = yes 
		c:SDK = {
			NOR = {
				has_modifier = resource_extraction_charter_modifier
				has_modifier = religious_missions_colonial_modifier
				has_modifier = settlement_campaigns_modifier
			}

		}
	}

    possible = {
		has_technology_researched = civilizing_mission
	}

    when_taken = {
		trigger_event = {
			id = dvg_japan.8
		}
    }

	ai_chance = {
		base = 100
	}
}

dvg_embrace_ainu = {
	is_shown = {
		any_scope_state = {

			state_region = {

				is_homeland = ainu
			}
			literacy_rate >= 0.80 
			
		}
	}

    possible = {
		any_interest_group = {
			is_powerful = yes
			law_stance = {
				law = law_type:law_cultural_exclusion
				value > neutral
			}
		}
	}

    when_taken = {
		add_primary_culture = cu:ainu
    }

	ai_chance = {
		base = 100
	}
}
dvg_ditch_shinto = {
	is_shown = {
		country_has_state_religion = rel:shinto
	}

    possible = {
		
		NOT = {
			has_law = law_type:law_theocracy
			has_law = law_type:law_monarchy
		}
	}

    when_taken = {
		set_state_religion = rel:mahayana
    }

	ai_chance = {
		base = 100
	}
}
