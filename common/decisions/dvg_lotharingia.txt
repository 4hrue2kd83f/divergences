dvg_odonia_unite = {
	is_shown = {

		this = c:ODO
		
	}

    possible = {
		 
		has_state_in_state_region = STATE_LORRAINE_ATLANTIQUE
		
		NOT = { exists = c:LOT }
		
		war = no
	
	}

    when_taken = {
		trigger_event = { id = dvg_odonia.1 popup = yes }
    }

	ai_chance = {
		base = 100
	}
}
