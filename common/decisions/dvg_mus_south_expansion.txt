mus_southern_expansion = {
	is_shown = {
		exists = c:MUS
		root = c:MUS
		NOT = { has_variable = southern_expansion }
	}

    possible = {
		has_technology_researched = nationalism
	}

    when_taken = {
        if = {
			limit = {
				owns_entire_state_region = STATE_DANU
				owns_entire_state_region = STATE_ROSTOV
				owns_entire_state_region = STATE_IDEL
			}
			trigger_event = { id = mus_southern_expansion.1 }
		}
		else = {
			add_journal_entry = {
				type = je_mus_southern_expansion
			}
		}
		ig:ig_landowners = {
			add_modifier = {
				name = ig_manifest_destiny
				months = normal_modifier_time
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = ig_manifest_destiny
				months = normal_modifier_time
				multiplier = -1
			}
		}
		# Add Claims
		s:STATE_DANU = { 
			if = {
				add_claim = ROOT 
			}
		}
		s:STATE_ROSTOV = { 
			if = {
				add_claim = ROOT 
			}
		}
		s:STATE_KURSK = { 
			if = {
				add_claim = ROOT 
			}
		}
		set_variable = southern_expansion
    }

	ai_chance = {
		base = 100
	}
}
