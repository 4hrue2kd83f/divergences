
dvg_annex_sahara = {
	is_shown = {
	
		NOT = { is_country_type = decentralized }

		
		any_neighbouring_state = {
			owner = {
				is_country_type = decentralized
			}
		}
		has_interest_marker_in_region = sr:region_north_africa
	
	}

    possible = {

		OR = {
			is_player = yes
			year >= 1910
			is_country_type = recognized
		}
		OR = {
		has_technology_researched = nationalism
		is_country_type = recognized
		}
		has_technology_researched = colonization
		NOT = { has_law = law_type:law_no_colonial_affairs }
		NOT = {
			has_modifier = decision_cooldown
		}
	
	}

    when_taken = {
		every_country = {
			if = {
				limit = {
					is_ai = yes			
					is_country_type = decentralized
					any_scope_state = {
						any_neighbouring_state = {
							owner = ROOT
						}
					}
				    country_is_in_africa = yes
				}			
				root = { annex = prev }
			}
		}
		create_incident = {
			country = ROOT
			value = 6
		}
		add_modifier = {
			name = decision_cooldown
			months = 60
		}
    }
	ai_chance = {
		base = 1
	}
	
}

dvg_maghreb_liberation = {
	is_shown = {
		
		
	NOT = {has_global_variable = maghreb_liberation}

	OR = {

	
	
	AND = {
		exists = c:MGB

		this = c:MGB
		}	
	
	country_has_primary_culture = cu:maghrebi
	country_has_primary_culture = cu:berber		
	
	}

		
	}

    possible = {
		has_technology_researched = pan-nationalism

		any_interest_group = {
				leader = {
					has_ideology = ideology:ideology_pacifist
				}
				is_powerful = no
			}
		country_rank >= rank_value:unrecognized_regional_power 
	}
		
		

    when_taken = {
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:maghrebi
				is_homeland = cu:berber
				}
				region = sr:region_north_africa
			}
			add_claim = ROOT
		}
		
		
		create_incident = {
			country = ROOT
			value = 15
		}
		set_global_variable = {
				name = maghreb_liberation
				value = 0
			}
    }

	ai_chance = {
		base = 100
	}
}
dvg_embrace_maghrebi = {
	is_shown = {
		any_primary_culture = {
			has_discrimination_trait = arab_speaking
		}
		NOT = {
			country_has_primary_culture = cu:maghrebi
		}
	}

    possible = {
		has_technology_researched = pan-nationalism
		cu:maghrebi = {
			NOT = {
			culture_is_discriminated_in = ROOT
			}
		}
		any_scope_state = {
			is_incorporated = yes
			state_region = {
				is_homeland = cu:maghrebi
			}
		}
		
	}

    when_taken = {
		add_primary_culture = cu:maghrebi
    }

	ai_chance = {
		base = 100
	}
}
