dvg_ven_patrician_decision = {
	is_shown = {
		exists = c:VEN
		c:VEN = THIS
		has_modifier = dvg_ven_patricians
	}

	possible = {
		OR = {
			AND = {
				ig:ig_landowners = {
					is_marginal = yes
					is_in_government = no
				}
				ig:ig_landowners = {
					is_marginal = yes
					is_in_government = no
				}
			}
		}
		NOT = { has_law = law_type:law_oligarchy }
	}

	when_taken = {
		if = {
			limit = {
				has_variable = dvg_ven_castes_preserved
			}
			ig:ig_landowners = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_venician_nuovo_patrician
			}
			ig:ig_petty_bourgeoisie = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_venician_nuovo_patrician
			}
		}
		if = {
			limit = {
				has_variable = dvg_ven_castes_abolished
			}
			ig:ig_landowners = {
				remove_ideology = ideology_venician_patrician
			}
			ig:ig_petty_bourgeoisie = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_reactionary
			}
		}
		remove_modifier = dvg_ven_patricians
		
	}

	ai_chance = {
		base = 100
	}
}
