dvg_ven_reform_patrician_system_decision = {
	is_shown = {
		exists = c:VEN
		c:VEN = THIS
		has_modifier = dvg_ven_patricians
	}

	possible = {
		OR = {
			AND = {
				ig:ig_landowners = {
					is_powerful = no
					is_in_government = no
				}
				ig:ig_ig_petty_bourgeoisie = {
					is_powerful = no
					is_in_government = no
				}
			}
		}
		NOT = { has_law = law_type:law_oligarchy }
	}

	when_taken = {
		if = {
			limit = {
				has_variable = dvg_ven_patrician_victory
			}
			ig:ig_landowners = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_venician_nuovo_patrician
				add_modifier = {
					name = dvg_ven_patricians_diluted_power
					months = 24
				}
			}
			ig:ig_petty_bourgeoisie = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_venician_nuovo_patrician
				add_modifier = {
					name = dvg_ven_patricians_diluted_power
					months = 24
				}
			}
		}
		if = {
			limit = {
				has_variable = dvg_ven_revolutionary_victory
			}
			ig:ig_landowners = {
				remove_ideology = ideology_venician_patrician
				add_modifier = {
					name = dvg_ven_patricians_diluted_power
					months = 24
				}
			}
			ig:ig_petty_bourgeoisie = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_reactionary
				add_modifier = {
					name = dvg_ven_patricians_diluted_power
					months = 24
				}
			}
		}
		remove_modifier = dvg_ven_patricians

	}

	ai_chance = {
		base = 100
	}
}
