	is_shown = {
		exists = c:VIN
		exists = c:SCA
		root = c:VIN
		c:SCA = {
			is_ai = yes
		}
		
		NOT = {
			has_variable = dvg_the_yolngu_war_hongshan_variable
		}
		c:VIN = {
			is_subject_of = c:SCA
		}
		is_diplomatic_play_committed_participant = no

	}

    possible = {
		has_technology_researched = nationalism
		c:SCA = {
			is_at_war = no
		}
		NOT = {has_truce_with = c:SCA}
		is_diplomatic_play_committed_participant = no
		liberty_desire > 75
	}

    when_taken = {
		
			if = {
				limit = {
					c:SCA = {
						is_ai = yes
					}
				}
				c:VIN = {
					make_independent = yes
				}
			}
			
		
    }

		base = 100
	}
}
