dvg_noongar_colony = {
	is_shown = {
		has_game_rule = australia_railroaded
		exists = c:NNG
		root = {
			has_state_in_state_region = STATE_NOONGAR
		}
		NOT = {
			country_has_primary_culture = cu:aborigine
		}
	}

    possible = {
		year > 1840
	}

    when_taken = {
		trigger_event = { id = dvg_oceania.3 popup = yes }
    }

	ai_chance = {
		base = 100
	}
}
