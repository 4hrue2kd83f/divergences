je_panjabi_revolt = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	

	possible = {
		
	}

	immediate = {
		set_variable = {
			name = punjab_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:panjabi = {
			 culture_is_discriminated_in = ROOT 
		}	
				}
				change_variable = {
					name = punjab_var
					add = 1
				}
			}
		}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		trigger_if = {
			limit = { exists = var:punjab_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		is_at_war = yes
		
		
		
	}
	
	on_complete = {
		set_variable = dvg_punjabi_revolt_complete
		trigger_event = {
			id = dvg_persia.2
		}
	}
	
	fail = {
		OR = {
				NOT = {owns_entire_state_region = STATE_PUNJAB}
				cu:panjabi = {
				NOT = { culture_is_discriminated_in = ROOT }
			}
		}
	}
	
	on_fail = {
		set_variable = dvg_panjabi_accepted
		
	}

	
	current_value = {
		value = root.var:punjab_var
	}

	goal_add_value = {
		value = 48
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
