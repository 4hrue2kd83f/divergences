je_seven_sisters = {
	icon = "gfx/interface/icons/invention_icons/nationalism.dds"

	group = je_group_crises

	immediate = {
		
	}

	is_shown_when_inactive = {
		country_has_primary_culture = cu:septiman
	}
	

	possible = {
		country_has_primary_culture = cu:septiman
	}

	on_yearly_pulse = {
		effect = {
			if = {
				limit = {
					has_technology_researched = empiricism
					NOT = {
						has_variable = constitutional_convention
					}
					exists = c:CNF
					c:CNF = ROOT
					is_subject = no
				}
				trigger_event = {
					id = dvg_belgium.5
				}
				set_variable = constitutional_convention

			}
			if = {
				limit = {
					has_technology_researched = nationalism
					NOT = {
						has_variable = belgian_conference
					}
					is_subject = yes
				}
				trigger_event = {
					id = dvg_belgium.8
				}
				set_variable = belgian_conference

			}
			
		}
		
	}

	complete = {
		custom_tooltip = {
			text = belgian_formed_tt
			AND = {
				exists = c:BLC
				c:BLC = ROOT
			}
		}
			
	}

	on_complete = {
		trigger_event = { id = dvg_belgium.7 }
		if = {
			limit = {
				exists = c:PLA
			}
			c:PLA = {
				trigger_event = { id = dvg_belgium.9 }

			}
		}
		set_global_variable = belgium_united
		call_election = {
			months = 1
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_technology_researched = nationalism
					NOT = {
						has_variable = belgian_destiny
					}
				}
				set_variable = belgian_destiny
				trigger_event = {
					id = dvg_belgium.8
				}
			}
		}
		random_events = {
			10 = dvg_belgium.41
		}
		
		
	}
	fail = {
		has_variable = constitutional_convention
		OR = {
			any_country = {
				count >= 3
				liberty_desire > 75
				country_has_primary_culture = cu:septiman
				is_subject_of = ROOT
				has_attitude = {
					who = ROOT
					attitude = rebellious
				}
			}
			NOT = {
				any_subject_or_below = {
					country_has_primary_culture = cu:septiman
				}
				any_scope_state = {
					count < 3
					state_region = {
						is_homeland = cu:septiman
					}
				}
			}
		}
		
		
		
	}
	on_fail = {
		trigger_event = {
			id = dvg_belgium.65
		}
	}
	
	should_be_pinned_by_default = yes
}

					NOT = {

					id = dvg_belgium.42

					}

						s:STATE_HAUTE_VALOISIE.region_state:BLC = {
							turmoil > 0.50
						}
						s:STATE_FLANDRE_SUPERIEURE.region_state:BLC = {
							turmoil > 0.50
						}
					}
					
					
				}
				trigger_event = {
					id = dvg_belgium.29
				}
			}
			if = {
			
				limit = {
					NOT = {
						exists = c:CHV
						exists = c:CFS
					}
					OR = {
						s:STATE_PHILIPPA.region_state:BLC = {
							turmoil > 0.50
						}
						s:STATE_CAROLINE.region_state:BLC = {
							turmoil > 0.50
						}
					}
					
					
				}
				trigger_event = {
					id = dvg_belgium.31
				}
			}

					NOT = {
						exists = c:CBV
					}

						turmoil > 0.50
					}

						is_incorporated = yes

							is_homeland = cu:septiman 
						}

					}
				}

					id = dvg_belgium.32
				}
			}
		}
		
	}

	}

	}

	}

	goal_add_value = 700




	
	should_be_pinned_by_default = yes
}
