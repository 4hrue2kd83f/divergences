je_seven_sisters = {
	icon = "gfx/interface/icons/invention_icons/nationalism.dds"

	group = je_group_crises

	immediate = {
		
	}

	is_shown_when_inactive = {
		country_has_primary_culture = cu:septiman
	}
	

	possible = {
		country_has_primary_culture = cu:septiman
	}

	on_yearly_pulse = {
		effect = {
			if = {
				limit = {
					has_technology_researched = empiricism
					NOT = {
						has_variable = constitutional_convention
					}
					exists = c:CNF
					c:CNF = ROOT
					is_subject = no
				}
				trigger_event = {
					id = dvg_belgium.5
				}
				set_variable = constitutional_convention

			}
			if = {
				limit = {
					has_technology_researched = nationalism
					NOT = {
						has_variable = belgian_conference
					}
					is_subject = yes
				}
				trigger_event = {
					id = dvg_belgium.8
				}
				set_variable = belgian_conference

			}
			
		}
		
	}

	complete = {
		custom_tooltip = {
			text = belgian_formed_tt
			AND = {
				exists = c:BLC
				c:BLC = ROOT
			}

		}
		
		
	}

	on_complete = {
		trigger_event = { id = dvg_belgium.7 }
		if = {
			limit = {
				exists = c:PLA
			}
			c:PLA = {
				trigger_event = { id = dvg_belgium.9 }

			}
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_technology_researched = nationalism
					NOT = {
						has_variable = belgian_destiny
					}
				}
				set_variable = belgian_destiny
				trigger_event = {
					id = dvg_belgium.8
				}
			}
		}
		
	}
	fail = {
		OR = {
			any_country = {
				count >= 3
				liberty_desire > 75
				country_has_primary_culture = cu:septiman
				is_subject_of = ROOT
				
			}
			NOT = {
				any_subject_or_below = {
					country_has_primary_culture = cu:septiman
				}
				country_tier = kingdom
				any_scope_state = {
					count < 3
					state_region = {
						is_homeland = cu:septiman
					}
				}
			}
		}
		
		
		
	}
	on_fail = {
		
	}
	
	should_be_pinned_by_default = yes
}

je_belgian_confederacy = {
	icon = "gfx/interface/icons/invention_icons/nationalism.dds"

	group = je_group_internal_affairs

	immediate = {
	}

	is_shown_when_inactive = {


	}
	

	possible = {
		
	}

	

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
	}

	on_complete = {
		
	}

	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					has_variable = recently_lost_war
					
				}
				change_variable = {
					name = dvg_belgian_var
					subtract = 1
				}
			}
			if = {
				limit = {
					has_modifier = declared_bankruptcy
				}
				change_variable = {
					name = dvg_belgian_var
					subtract = 1
				}
			}
			if = {
				limit = {
					any_interest_group = {
						is_insurrectionary = yes
					}
				}
				change_variable = {
					name = dvg_belgian_var
					subtract = 1
				}
			}
			if = {
				limit = {
					every_scope_state = {
						turmoil < 0.50
					}
				}
				change_variable = {
					name = dvg_belgian_var
					add = 1
				}
			}
			if = {
				limit = {
					owns_entire_state_region = STATE_SANACOMIE
				}
				change_variable = {
					name = dvg_belgian_var
					add = 1
				}
			}
		}
		
	}

	on_monthly_pulse = {
		effect = {
			if = {
			
				limit = {
					NOT = {
						exists = c:CHV
						exists = c:CFS
					}
					OR = {
						s:STATE_HAUTE_VALOISIE.region_state:BLC = {
							turmoil > 0.50
						}
						s:STATE_FLANDRE_SUPERIEURE.region_state:BLC = {
							turmoil > 0.50
						}
					}
					
					
				}
				trigger_event = {
					id = dvg_belgium.29
				}
			}
			if = {
			
				limit = {
					NOT = {
						exists = c:CHV
						exists = c:CFS
					}
					OR = {
						s:STATE_PHILIPPA.region_state:BLC = {
							turmoil > 0.50
						}
						s:STATE_CAROLINE.region_state:BLC = {
							turmoil > 0.50
						}
					}
					
					
				}
				trigger_event = {
					id = dvg_belgium.31
				}
			}
			if = {
			
				limit = {
					NOT = {
						exists = c:CBV
					}
					s:STATE_BASSE_VALOISIE.region_state:BLC = {
						turmoil > 0.50
					}
					any_scope_state = {
						owner = ROOT
						is_incorporated = yes
						state_region = {
							is_homeland = cu:septiman 
						}
						count < 4
					}
				}
				trigger_event = {
					id = dvg_belgium.32
				}
			}
		}
		
	}


	fail = {
		this = { var:dvg_belgian_var <= 0 }
	}

	on_fail = {
		if = {
			limit = {
				NOT = {
					exists = c:CNF
				}
				any_scope_state = {
					state_region = STATE_NOUVELLE_FLANDRE
					is_capital = yes
				}
			}
			change_tag = CNF
		}
		if = {
			limit = {
				exists = c:CNF
				c:CNF = {
					is_ai = yes
				}
			}
			annex = c:CNF
			change_tag = CNF
		}
		
	}

	invalid = {
		
	}
	current_value = {
		value = var:dvg_belgian_var
	}

	progressbar = yes
	goal_add_value = 700




	
	should_be_pinned_by_default = yes
}
