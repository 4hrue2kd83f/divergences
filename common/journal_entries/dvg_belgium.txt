je_seven_sisters = {
	icon = "gfx/interface/icons/invention_icons/nationalism.dds"

	group = je_group_crises

	immediate = {
		
	}

	is_shown_when_inactive = {
		country_has_primary_culture = cu:septiman
	}
	

	possible = {
		country_has_primary_culture = cu:septiman
	}

	on_yearly_pulse = {
		effect = {
			if = {
				limit = {
					has_technology_researched = empiricism
					NOT = {
						has_variable = constitutional_convention
					}
					exists = c:CNF
					c:CNF = ROOT
					is_subject = no
				}
				trigger_event = {
					id = dvg_belgium.5
				}
				set_variable = constitutional_convention

			}
			if = {
				limit = {
					has_technology_researched = nationalism
					NOT = {
						has_variable = belgian_conference
					}
					is_subject = yes
				}
				trigger_event = {
					id = dvg_belgium.8
				}
				set_variable = belgian_conference

			}
			
		}
		
	}

	complete = {
		OR = {

			AND = {
			owns_entire_state_region = STATE_CAROLINE
			owns_entire_state_region = STATE_PHILIPPA
			owns_entire_state_region = STATE_NOUVELLE_LORRAINE
			owns_entire_state_region = STATE_NOUVELLE_FLANDRE 
			owns_entire_state_region = STATE_HAUTE_VALOISIE
			owns_entire_state_region = STATE_BASSE_VALOISIE
			owns_entire_state_region = STATE_FLANDRE_SUPERIEURE
			}
		}
		
	}

	on_complete = {
		if = {
			limit = {
				exists = c:PLA
			}
			c:PLA = {
				trigger_event = { id = dvg_belgium.9 }

			}
		}
	}
	on_monthly_pulse = {
		
	}
	fail = {
		NOT = {
			any_subject_or_below = {
				country_has_primary_culture = cu:septiman
			}
			any_country = {
				count > 3
				country_has_primary_culture = cu:septiman
				is_subject = no
				relations:root <= relations_threshold:cold
			}
		}
		
		
	}
	
	should_be_pinned_by_default = yes
}

	}
	

	possible = {

	}

	on_complete = {
		trigger_event = { id = dvg_belgium.27 }	
	}
	on_monthly_pulse = {
		random_events = {

	}

	}
	
	should_be_pinned_by_default = yes
