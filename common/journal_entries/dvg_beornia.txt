je_cattle_barons_wrath = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	

	possible = {
		
	}

	immediate = {
		set_variable = {
			name = cattle_barons_var
			value = 0
		}
	}

	

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		trigger_if = {
			limit = { exists = var:cattle_barons_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		
		
		
	}
	
	on_complete = {
		set_variable = house_of_cards_complete
		trigger_event = {
			id = dvg_beornia.5
		}
	}
	
	fail = {
		OR = {
			ig:ig_landowners = {
				is_powerful = no
				is_in_government = no
			}
			NOT = { 
			        has_law = law_type:law_monarchy
			}
			
			
		}
	}
	
	on_fail = {
		set_variable = house_of_cards_complete
		trigger_event = {
			id = dvg_beornia.6
		}
	}
	

	

	current_value = {
		value = root.var:cattle_barons_var
	}

	goal_add_value = {
		value = 8
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
