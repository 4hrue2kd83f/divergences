	}

	on_complete = {

			id = dvg_burgundy.34
		}
	}

		}
		var:dvg_je_an_var = {
			value <= 0
		}
	}
	on_fail = {
		#Palriamentary Victory
		has_global_variable = jean_coup

			id = dvg_burgundy.32
		}
	}
	current_value = {
		value = var:dvg_je_an_var
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					ig:ig_landowners = {
						is_powerful = yes
						is_in_government = yes
						ig_approval >= 0 
					}

					has_variable = happy_landowners
				}
				}
				change_variable = {
					name = dvg_je_an_var
					add = 2
				}
				set_variable = happy_landowners
			}
			if = {
				limit = {
					ig:ig_landowners = {
						is_powerful = no
						is_in_government = no
						ig_approval >= 0 
					}
				has_variable = happy_landowners
				
				}
				change_variable = {
					name = dvg_je_an_var
					subtract = 2
				}
				remove_variable = happy_landowners
			}
			if = {
				limit = {
					ig:ig_intelligentsia = {
						is_powerful = yes
						is_in_government = yes
						ig_approval >= 0 
					}
				NOT = {
					has_variable = happy_intelligentsia
				}
				}
				change_variable = {
					name = dvg_je_an_var
					subtract = 2
				}
				set_variable = happy_intelligentsia
			}
			if = {
				limit = {
					ig:ig_intelligentsia = {
						is_powerful = yes
						is_in_government = yes
						ig_approval >= 0 
					}
					has_variable = happy_intelligentsia

				}
				change_variable = {
					name = dvg_je_an_var
					add = 2
				}
				remove_variable = happy_intelligentsia
			}
			if = {
				limit = {
					NOT = {
						has_global_variable = jean_coup
					}
					year >= 1855 
					OR = {
						var:dvg_je_an_var = {
							value >= 10
						}
						var:dvg_je_an_var = {
							value <= 0
						}
						any_interest_group = {
							is_marginal = no
							is_insurrectionary = yes
						}
					}

				}
				trigger_event = {
					id = dvg_burgundy.31
				}
				
			}

			
		}


	}
	invalid = {
		NOT = {
			any_scope_character = {
				has_ideology = ideology:ideology_jeanist_leader
			}
		}
	}
	on_invalid = {
		trigger_event = {
			id = dvg_burgundy.36
		}
	}
	goal_add_value = {
		add = 10
	}
	timeout = 3650
	on_timeout = {

			id = dvg_burgundy.35
		}
		
	}

je_unify_brg = {
	icon = "gfx/interface/icons/event_icons/event_empty_chair.dds"

	group = je_group_crises

	

	is_shown_when_inactive = {

		
	}

	possible = {	
		exists = c:BRG
		exists = c:PBG
	}
	immediate = {
		hidden_effect = {
			
			if = {
				limit = {
					c:BRG = ROOT
				}
				set_secret_goal = {
					country = c:PBG
					secret_goal = conquer
				}
			}
		}
	}

	complete = {
			custom_tooltip = {
				text = burgundy_united_tt
				NOT = {
					exists = c:PBG
				}
			}
	}

	on_complete = {
		ig:ig_petty_bourgeoisie = {
			remove_ideology = ideology_meritocratic
			add_ideology = ideology_jeanist
		} 
		ig:ig_landowners = {
			remove_ideology = ideology_stratocratic
			add_ideology = ideology_jeanist
		} 
		add_modifier = {
			name = dvg_modifier_landowners_bolstered
			months = 120
			is_decaying = yes
		}
	}

	fail = {
		NOT = {
			any_character = {
				NOT = {
					has_ideology = ideology:ideology_jeanist_leader
				}
			}
		}
	}
	on_fail = {
		
	}

	timeout = 7300

	on_timeout = {
		if = {
			limit = {
				any_scope_state = {
					is_capital = yes
					state_region = {
						is_homeland = cu:dutch
					}
				}
				NOT = {
					exists = c:NET
				}
			}
			change_tag = NET
		}
		if = {
			limit = {
				exists = s:STATE_BURGUNDY.region_state:BRG
				s:STATE_BURGUNDY.region_state:BRG = {
					is_capital = yes
				}
				NOT = {
					exists = c:BGY
				}
			}
			change_tag = BGY
		}
	}
	should_be_pinned_by_default = yes
	inheritable = no
}
