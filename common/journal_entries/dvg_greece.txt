je_dvg_greek_nationalism = {
	icon = "gfx/interface/icons/invention_icons/nationalism.dds"
	
	group = je_group_historical_content

	is_shown_when_inactive = {
		any_scope_state = {
			state_region = {
				is_homeland = cu:greek
			}
		}
		#exists = c:VEN
		#c:VEN = ROOT
		any_scope_character = {
			has_variable = zimisces_var
		}
		
		NAND = {
			c:GRE = ROOT
			exists = c:GRE
		}
	}

	possible = {
		OR = {
			has_technology_researched = nationalism
			AND = {
				exists = c:GRE
				c:GRE = {
					has_technology_researched = nationalism
				}
			}
			

		}
	}

	immediate = {
		if = {
			limit = {
				any_scope_character = {
					has_variable = zimisces_var
				}
				exists = c:VEN
				c:VEN = ROOT
				
			}
			trigger_event = {
				id = dvg_greece.201
			}

		}
		set_variable = {
			name = greece_var
			value = 0
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:greek = {
			 culture_is_discriminated_in = ROOT
					
			}	
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					any_scope_state = {
						state_region = {
							is_homeland = cu:greek
						}
						turmoil >= .30 
					}
					
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					NOT = {
						any_country = {
							country_has_primary_culture = cu:greek
						}
					}
					
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					any_subject_or_below = {
						country_has_primary_culture = cu:greek
						liberty_desire > 30

					}
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					any_subject_or_below = {
						country_has_primary_culture = cu:greek
						liberty_desire > 50
					}
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					cu:greek = {
				culture_secession_progress = {
					target = ROOT
					value > 0.5
				}	
				}
				
						}
				change_variable = {
					name = greece_var
					add = 24
				}
			}
			if = {
				limit = {
					NOT = {
						exists = c:GRE
						has_variable = greece_to_liberate
					}
				}
				set_variable = greece_to_liberate

				
			}
			if = {
				limit = {
					has_variable = council_start
					NOT = {
						has_variable = greece_citizen_var
					}
				}
				trigger_event = {
					id = dvg_greece.210
				}
				
			}
			if = {
				limit = {
					has_variable = council_start
					NOT = {
						has_variable = greece_voting_var
					}
				}
				trigger_event = {
					id = dvg_greece.211
				}
			}
			if = {
			#Greek Integration

				limit = {
					has_variable = council_start
					has_variable= greece_citizen_var
					OR = {
						has_variable = greece_voting_var
						has_law = law_type:law_census_voting
					}	
					OR = {
						has_variable= greece_citizen_var
						NOT = {
							cu:greek = {
								culture_is_discriminated_in = ROOT
							}
						}
					}
					
				}
				trigger_event = {
					id = dvg_greece.212
				}
			}
			#Greek Revolt
			if = {
				limit = {
					has_variable = council_start
					has_variable = greece_voting_var
					has_variable= greece_citizen_var
					OR = {
						cu:greek = {
							culture_is_discriminated_in = ROOT
						}
						country_has_voting_franchise = no
						
					}
					any_subject_or_below = {
						country_has_primary_culture = cu:greek
						liberty_desire > 65
					}	
				}
				trigger_event = {
					id = dvg_greece.203
				}
			}
		}
	}
	

	complete = {
		
	}

	on_complete = {
		if = {
			limit = {
				exists = c:GRE
				has_state_in_state_region = STATE_CYPRUS
			}
			trigger_event = {
				id = dvg_greece.203
			}
		}
	}
	invalid = {
		OR = {
			NAND = {
				any_scope_state = {
					state_region = {
						is_homeland = cu:greek
					}
				}
				any_subject_or_below = {
					country_has_primary_culture = cu:greek
				}
			}
			has_variable = council_concluded
			is_diplomatic_play_enemy_of = c:GRE

		}
		
	

	}
	fail = {
		
	}

	weight = 500
	should_be_pinned_by_default = yes
	progressbar = yes
	goal_add_value = {
		value = 48
	}
	
}

	icon = "gfx/interface/icons/event_icons/event_military.dds"
	
	group = je_group_historical_content

	is_shown_when_inactive = {
		exists = c:GRE
		exists = c:VEN
		c:GRE = ROOT
		any_scope_character = {
			has_variable = zimisces_var
		}
	}

	possible = {
		is_diplomatic_play_enemy_of = c:VEN


	}

	immediate = {
		
	}
	
	complete = {
		is_diplomatic_play_committed_participant = no
		is_subject = no
	}

	on_complete = {
		c:VEN = {
			trigger_event = {
				id = dvg_greece.209
			}
		}
		if = {
			limit = {
				any_scope_character = {
					has_variable = zimisces_var
				}
			}
			trigger_event = {
				id = dvg_greece.214 
			}
		}
		
	
	}
	fail = {
		OR = {
			is_subject = yes
			NOT = {
				exists = ROOT
			}
		}
		NOT = {
			owns_entire_state_region = s:STATE_PELOPONNESE
			owns_entire_state_region = s:STATE_THESSALIA
			owns_entire_state_region = s:STATE_CYPRUS
			owns_entire_state_region = s:STATE_ATTICA
		}
	}
	on_fail = {
		c:VEN = {
			trigger_event = {
				id = dvg_greece.216
			}
		}
	}
	invalid = {
		NOT = {
			exists = c:VEN
		}
	}

	weight = 20
	should_be_pinned_by_default = yes
}
