je_dvg_greek_nationalism = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	
	group = je_group_historical_content

	is_shown_when_inactive = {
		any_scope_state = {
			state_region = {
				is_homeland = cu:greek
			}
		}
		#exists = c:VEN
		#c:VEN = ROOT
		any_scope_character = {
			has_variable = zimisces_var
		}
	}

	possible = {
		has_technology_researched = nationalism
	}

	immediate = {
		if = {
			limit = {
				any_scope_character = {
					has_variable = zimisces_var
				}
				exists = c:VEN
				c:VEN = ROOT
				
			}
			trigger_event = {
				id = dvg_greece.201
			}

		}
		set_variable = {
			name = greece_var
			value = 0
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:greek = {
			 culture_is_discriminated_in = ROOT
					
			}	
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					any_scope_state = {
						state_region = {
							is_homeland = cu:greek
						}
						turmoil >= .30 
					}
					
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					NOT = {
						any_country = {
							country_has_primary_culture = cu:greek
						}
					}
					
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					any_subject_or_below = {
						country_has_primary_culture = cu:greek
					}
					liberty_desire > 33
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					cu:greek = {
				culture_secession_progress = {
					target = ROOT
					value > 0.5
				}	
				}
				
						}
				change_variable = {
					name = greece_var
					add = 24
				}
			}
			if = {
				limit = {
					NOT = {
						exists = c:GRE
						has_variable = greece_to_liberate
					}
				}
				set_variable = greece_to_liberate

			}
		}
	}
	

	complete = {
		
	}

	on_complete = {
		if = {
			limit = {
				exists = c:GRE
				has_state_in_state_region = STATE_CYPRUS
			}
			trigger_event = {
				id = dvg_ven.202
			}
		}
	}
	invalid = {
		NOT = {
			any_scope_state = {
				state_region = {
					is_homeland = cu:greek
				}
			}
		}
		NOT = {
			any_subject_or_below = {
				country_has_primary_culture = cu:greek
			}
		}
	}

	weight = 500
	should_be_pinned_by_default = yes
	progressbar = yes
	goal_add_value = {
		value = 48
	}
	
}
