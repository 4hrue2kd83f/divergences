je_greek_nationalism = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	
	group = je_group_historical_content

	is_shown_when_inactive = {
		any_scope_state = {
			state_region = {
				is_homeland = cu:greek
			}
		}
		
	}

	possible = {
		has_technology_researched = nationalism
	}

	immediate = {
		if = {
			limit = {
				any_scope_character = {
					has_variable = zimisces_var
				}
				exists = c:VEN
				c:VEN = ROOT
				
			}
			trigger_event = {
				id = dvg_ven.201
			}

		}
		set_variable = {
			name = greece_var
			value = 0
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:greek = {
			 culture_is_discriminated_in = ROOT 
			}	
				}
				change_variable = {
					name = greece_var
					add = 1
				}
			}
			if = {
				limit = {
					cu:greek = {
				culture_secession_progress = {
					target = ROOT
					value > 0.5
				}	
				}
				
						}
				change_variable = {
					name = greece_var
					add = 24
				}
			}
		}
	}
	

	complete = {
		
	}

	on_complete = {
		if = {
			limit = {
				exists = c:GRE
				has_state_in_state_region = STATE_CYPRUS
			}
			trigger_event = {
				id = dvg_ven.202
			}
		}
	}

	weight = 20
	should_be_pinned_by_default = yes
	progressbar = yes
	goal_add_value = {
		value = 48
	}
}
