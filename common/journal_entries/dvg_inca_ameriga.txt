je_beornia_revolt = {
	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	group = je_group_internal_affairs

	immediate = {
		set_variable = {
			name = beornia_hope_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:beornian = {
					 culture_is_discriminated_in = ROOT 
					}	
				}
				change_variable = {
					name = armenia_var
					add = 1
				}
			}
			if = {
				limit = {
					cu:beornian = {
						culture_secession_progress = {
							target = ROOT
							value > 0.9
						}	
					}
				}
				change_variable = {
					name = armenia_var
					add = 24
				}
			}
		}
	}

	complete = {
		trigger_if = {
			limit = { exists = var:beornia_hope_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
	}

	goal_add_value = {
		value = 100
	}

    on_complete = {
		trigger_event = { id = dvg_hbe.4 }
	}

	progressbar = yes
	weight = 100
	should_be_pinned_by_default = yes
}

je_inca_lotharingia = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_foreign_affairs

	complete = {
		is_at_war = no
		OR = {
			custom_tooltip = {
				text = lot_existence_tt
				NOT = {exists = c:LOT}
			}
			AND = {
				c:LOT = { is_subject_of = ROOT }
			}
		}
	}
	
	on_complete = {
		set_country_type = recognized
		
		trigger_event = { id = dvg_inca_ameriga.5} 

		set_global_variable = lotharingia_shatters
	}

	fail = {
		OR = { 
			custom_tooltip = {
				text = twt_capitulation_tt
				has_variable = recent_capitulation
			}
			has_truce_with = c:LOT
		}
	}

	on_fail = {

		trigger_event = { id = dvg_inca_ameriga.9 }

		every_diplomatic_play = {
			limit = {
				initiator_is = c:TWT
			}
			end_play = yes
		}
	}

	timeout = 1095

	on_timeout = {
		trigger_event = { id = dvg_inca_ameriga.8 }
	}

	weight = 100
	should_be_pinned_by_default = yes
}
