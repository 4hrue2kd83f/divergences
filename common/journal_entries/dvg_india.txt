je_dvg_india_struggle = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	is_shown_when_inactive = {
		any_scope_state = {
			OR = {
				region = sr:region_north_india
				region = sr:region_south_india
			}
		}
	}
	possible = {
		any_country = {
			any_primary_culture = {
				has_discrimination_trait = south_asian_heritage 
			}
			has_technology_researched = nationalism
		}
		OR = {
			any_scope_state = {
				OR = {
					region = sr:region_north_india
					region = sr:region_south_india
				}
			}
			any_subject_or_below = {
				any_scope_state = {
					OR = {
						region = sr:region_north_india
						region = sr:region_south_india
					}
				}
			}
		}	
	}
	immediate = {
		set_global_variable = {
			name = india_struggle_var
			value = 0
		}
	}
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	on_complete = {
		trigger_event = {
			id = dvg_india_struggle.1
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = { any_primary_culture = { has_discrimination_trait = south_asian_heritage }}
					OR = {
						any_scope_state = {
							OR = {
								region = sr:region_north_india
								region = sr:region_south_india
							}
						}
						any_subject_or_below = {
							any_scope_state = {
								OR = {
									region = sr:region_north_india
									region = sr:region_south_india
									}
							}
						}
					}
				}
				change_global_variable = {
					name = india_struggle_var
					add = 1
				}
			}
			if = {
				limit = {
					NOT = { any_primary_culture = { has_discrimination_trait = south_asian_heritage }}
					any_scope_state = {
						OR = {
							region = sr:region_north_india
							region = sr:region_south_india
						}
						count >= 3
					}
				}
				change_global_variable = {
					name = india_struggle_var
					add = 1
				}
			}
		}
	}
	current_value = {
		value = global_var:india_struggle_var
	}
	goal_add_value = {
		add = 1080
	}
	progressbar = yes
	should_be_pinned_by_default = yes
}
