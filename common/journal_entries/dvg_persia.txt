je_armenian_revolt = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	

	possible = {
		
	}

	immediate = {
		set_variable = {
			name = armenia_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:armenian = {
			 culture_is_discriminated_in = ROOT 
		}	
				}
				change_variable = {
					name = armenia_var
					add = 1
				}
			}
		}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		trigger_if = {
			limit = { exists = var:armenia_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		
		
		
	}
	
	on_complete = {
		set_variable = armenia__revolt_complete
		trigger_event = {
			id = dvg_persia.2
		}
	}
	
	fail = {
		OR = {
			cu:armenian = {
			NOT = { culture_is_discriminated_in = ROOT }
		}
		}
	}
	
	on_fail = {
		set_variable = armenian_accepted
		trigger_event = {
			
		}
	}


	current_value = {
		value = root.var:armenia_var
	}

	goal_add_value = {
		value = 48
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
