je_armenian_revolt = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	

	possible = {
		
	}

	immediate = {
		set_variable = {
			name = armenia_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:armenian = {
			 culture_is_discriminated_in = ROOT 
		}	
				}
				change_variable = {
					name = armenia_var
					add = 1
				}
			}
		}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		trigger_if = {
			limit = { exists = var:armenia_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		
		
		
	}
	
	on_complete = {
		set_variable = armenia_revolt_complete
		trigger_event = {
			id = dvg_persia.2
		}
	}
	
	fail = {
		OR = {
			cu:armenian = {
			NOT = { culture_is_discriminated_in = ROOT }
		}
		}
	}
	
	on_fail = {
		set_variable = armenian_accepted
		trigger_event = {
			
		}
	}


	current_value = {
		value = root.var:armenia_var
	}

	goal_add_value = {
		value = 48
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}

je_karkiya_rule = {
	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	is_shown_when_inactive = {
		
	}

	possible = {
	
	}
	
	complete = {
	
	OR ={
		ig:ig_intelligentsia = {
				is_marginal = no
				ig_approval < 0
			}
		ig:ig_intelligentsia = {
				is_insurrectionary = yes
			
			}	
		}
		taking_loans = yes
		
		NOT = {has_law = law_type:law_census_voting}
		
		root.average_sol < 8
	}
	
	on_complete = {
		trigger_event = { id = dvg_persia.10}
	}

	fail = {
		root.average_sol > 13
		ig:ig_intelligentsia = {
				is_marginal = yes
				
			}
	}

	on_fail = {
		trigger_event = { id = dvg_persia.13}
	}
	timeout = {
		10950
	}

	on_timeout = {
		trigger_event = { id = dvg_persia.13}
	}
	
	invalid = {
	
	
		
	}

	weight = 10000
	should_be_pinned_by_default = yes
}
