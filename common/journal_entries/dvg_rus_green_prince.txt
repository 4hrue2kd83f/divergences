je_rus_green_prince_nzh = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	# note that if you change this, also update the decision
	complete = {
        OR = {
            custom_tooltip = {
                text = green_prince_heir
                any_scope_character = {
                    is_heir = yes
                }
            }
        }
	}	
    
    on_complete = {
		custom_tooltip = green_prince_succession
		if = {
			limit = {
				exists = c:NZH
				c:NZH = ROOT
			}
			trigger_event = { id = dvg_green_prince_events.1 }
		}
	}

    invalid = {
		OR = {
			NOT = {
				exists = c:NZH
			}
            custom_tooltip = {
                text = green_prince_died
                ruler = {
                    NOT = { has_variable = is_green_prince }
                }
            }
        }
    }     

	weight = 10
	should_be_pinned_by_default = yes
}
