	}

		AND = {
			any_scope_state = {
				state_region = s:STATE_YUCATAN
				turmoil < 0.2
			}
			any_scope_state = {
				state_region = s:STATE_CHIAPAS
				turmoil < 0.2
			}
			any_scope_state = {
				state_region = s:STATE_GUATEMALA
				turmoil < 0.2
			}
			any_scope_state = {
				state_region = s:STATE_MANAGUA
				turmoil < 0.2
			}
			any_scope_state = {
				state_region = s:STATE_COSTA_RICA

			}
		}
		year < 1873
	}
	fail = {
		AND = {
			any_scope_state = {
				state_region = s:STATE_YUCATAN
				turmoil >= 0.2
			}
			any_scope_state = {
				state_region = s:STATE_CHIAPAS
				turmoil >= 0.2
			}
			any_scope_state = {
				state_region = s:STATE_GUATEMALA
				turmoil >= 0.2
			}
			any_scope_state = {
				state_region = s:STATE_MANAGUA
				turmoil >= 0.2
			}
			any_scope_state = {
				state_region = s:STATE_COSTA_RICA
				turmoil >= 0.2
			}
		}
		year = 1873
	}

	on_complete = {
		set_variable = { name = successful_marquez_gambit }
	}
	
	on_fail = {

	}
