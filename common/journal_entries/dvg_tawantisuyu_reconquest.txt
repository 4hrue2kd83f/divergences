je_llapa_succession = {
	icon = "gfx/interface/icons/invention_icons/event_newspaper.dds"

	#Completes just by making Intelligentsia strong enough, Llapa is the designated heir
	complete = {
		AND = {
			ig:ig_intelligentsia = { is_powerful = yes }
			ig:ig_armed_forces = { is_powerful = no }
		}
	}

    on_complete = {
		trigger_event = { id = dvg_tawantisuyu_reconquest.6 }
	}

	#If Intelligentsia haven't managed to gain power by 1838, then Washkar can take over
	fail = {
		AND = {
			ig:ig_intelligentsia = { is_powerful = no }
			game_date >= 1838.1.1
		}
	}
	on_fail = {
		trigger_event = { id = dvg_tawantisuyu_reconquest.7 }
	}
	invalid = {
	NOT = {
		any_character = {
					has_variable = is_llapa_inka
				}
		}
	}
	
	weight = 100
	should_be_pinned_by_default = yes
}

				}
				change_variable = {
					name = armenia_var

				}
			}
			if = {
				limit = {
					cu:beornian = {
				culture_secession_progress = {
					target = ROOT

				}	
				}
				
						}

					name = armenia_var
					add = 24
				}
			}
		}
	}

		trigger_if = {

	}

    on_complete = {
		trigger_event = { id = dvg_hbe.5 }
	}

	progressbar = yes
	weight = 100
	should_be_pinned_by_default = yes
}
