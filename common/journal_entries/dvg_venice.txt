je_ven_ease_demands = {
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		set_variable = {
			name = 
		}
	}

	fail = {
		custom_tooltip = {
			text = dua_global_revolted_tooltip
			OR = {
				has_global_variable = dua_global_revolted
				AND = {
					is_diplomatic_play_committed_participant = no
					is_at_war = no
					has_revolution = no
					OR = {
						exists = c:ENG
						exists = c:FRA
					}
				}
			}
		}
	}
	
	on_timeout = {
		set_global_variable = dua_timeout
	}

	on_monthly_pulse = {
		random_events = {
			30 = 0
			20 = dvg_ven.101 # Patron
			20 = dvg_ven.102 # Publication of "Histoire de la France depuis ses origines"
			20 = dvg_ven.103 # Publication of "Saxon Poetry"
			20 = dvg_ven.104 # The Great Travel of the Dauphin
			20 = dvg_ven.105 # The 'Immortals'
			20 = dvg_ven.106 # Radical IG
			20 = dvg_ven.107 # Celebration of the German Revolution
			30 = dvg_ven.108 # Springtime 1848
			30 = dvg_ven.109 # Valois subversion
			30 = dvg_ven.110 # Scottish subversion
			50 = dvg_ven.111 # Anglois resilience 1855
			50 = dvg_ven.112 # Anglois resilience 1860 

		}
	}

	current_value = {
		value = root.var:ven_ease_demands
	}

	goal_add_value = {
		add = 5
	}

	progressbar = yes

	timeout = 1825

	weight = 10000
	should_be_pinned_by_default = yes
}
