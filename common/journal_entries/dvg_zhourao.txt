			has_discrimination_trait = sinosphere
		}
	}

	possible = {
		has_technology_researched = civilizing_mission
		country_rank >= rank_value:minor_power
		average_sol >= 10
		is_at_war = no
		is_subject = no
		OR = {
			has_law = law_type:law_racial_segregation
			has_law = law_type:law_cultural_exclusion

		}
		NOR = {
			has_law = law_type:law_closed_borders
			has_law = law_type:law_isolationism
		}
	}

		set_variable = {
			name = zhourao_incorporated_population
			value = 0
		}
		set_variable = {
			name = zhourao_incorporated_states
			value = 0
		}
		every_scope_state = {
			limit = {
				is_incorporated = yes
				state_is_in_zhourao = yes

			ROOT = {
				change_variable = {
					name = zhourao_incorporated_population
					add = PREV.state_population
				}
				change_variable = {
					name = zhourao_incorporated_states
					add = 1
				}
			}
		}
		set_variable = {
			name = zhourao_migration_goal
			value = var:zhourao_incorporated_states
		}
		change_variable = {
			name = zhourao_migration_goal
			multiply = 1500000
		}
		trigger_event = { id = sa_migration.2 popup = yes }
	}

	on_weekly_pulse = {
		effect = {
			set_variable = {
				name = zhourao_incorporated_population
				value = 0
			}
			every_scope_state = {
				limit = {
					is_incorporated = yes
					
				}
				ROOT = {
					change_variable = {
						name = zhourao_incorporated_population

				name = zhourao_incorporated_states
				value = 0
			}
			every_scope_state = {
				limit = {
					is_incorporated = yes

				}
				ROOT = {
					change_variable = {
						name = zhourao_incorporated_states
						add = 1

				}

				name = zhourao_migration_goal
				value = var:zhourao_incorporated_states

				name = zhourao_migration_goal
				multiply = 1500000
			}
		}
	}

	complete = {

			text = sa_migration_completion_tt
