# root = The diplomatic play that has been started
# scope:initiator = Initiator of the play
# scope:target = Target of the play

on_diplomatic_play_started = {

	
	effect = {
		### Burgundy cant attack HRE
		if = {
			limit = {
				scope:initiator = {
					has_journal_entry = je_treaty_of_paris 
					this = c:BRG 
				}
			
				scope:target = { has_variable = dvg_hre_member }
			}
		

			root = { save_scope_as = dvg_brg_violate_treaty_of_paris_play }
			scope:initiator = { trigger_event = dvg_treaty_paris.1 }

		}

		}


	

		### Bohemia cant attack HRE
