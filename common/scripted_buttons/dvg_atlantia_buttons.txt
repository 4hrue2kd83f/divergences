
atlantia_subject_button = {
	name = "unite_atlantia_button"
	desc = "unite_atlantia_button_desc"

	visible = { 
		
		any_primary_culture = {
			has_discrimination_trait = amerigan
		}
		
	}

	possible = {

		any_subject_or_below = {

			any_primary_culture = {
				has_discrimination_trait = amerigan
			}

		}
		any_country = {

			any_primary_culture = {
				has_discrimination_trait = amerigan
			}
			owes_obligation_to = ROOT
			NOT = {
				has_truce_with = ROOT
			}
		
		}
		
		
	}
	effect = {
		every_country = {
			if = {
				limit = {
					is_ai = yes			
					any_primary_culture = {
						has_discrimination_trait = amerigan
					}
					owes_obligation_to = ROOT

				}	
				save_scope_as = atl_subject		
			}
		}
		create_diplomatic_pact = {
			country = scope:atl_subject
			type = protectorate
		}
		remove_diplomatic_pact = {
			country = scope:atl_investment
			type = bankroll
		}
		create_incident = {
			country = ROOT
			value = 6.0
		}
		
		
	}

	ai_chance = {
		base = 100
		 		 
	 }	
}
atlantia_investment_button = {
	name = "atl_invest_button"
	desc = "atl_invest_button_desc"

	visible = { 
		country_has_primary_culture = cu:atlantian
	}

	possible = {
		has_technology_researched = railways	
		has_technology_researched = banking
		any_neighbouring_state = {
			owner = {
				any_primary_culture = {
					has_discrimination_trait = amerigan
				}
				market = {
					owner = ROOT
				}
				NOT = {
					has_truce_with = ROOT
				}
			}
		}
	}
	effect = {

		every_country = {
			limit = {
				any_primary_culture = {
					has_discrimination_trait = amerigan
				}
				market = {
					owner = ROOT
				}
			}
			create_truce = {
				country = ROOT
				months = 72
			}
			save_scope_as = atl_investment
		}
		every_state = {
			limit = {
				state_region = {
					is_homeland = cu:atlantian
				}
			}
			create_building = {
				building = building_railway
			}
		}
		create_diplomatic_pact = {
			country = scope:atl_investment
			type = bankroll
		}
		
		
	}

	ai_chance = {
		base = 100
		 		 
	 }	
}

atlantia_agitation_button = {

	}

		has_technology_researched = mass_communication

			any_neighbouring_state = {
				owner = c:AMA

				owner = c:LOT
			}
		}
		NOT = {

	}

		add_modifier = {
			name = dvg_atlantia_giving_support
		}
		trigger_event = dvg_atlantia.18
	}

	ai_chance = {
		base = 100
		 		 
	 }	
}
