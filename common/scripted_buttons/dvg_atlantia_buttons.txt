
atlantia_subject_button = {
	name = "unite_atlantia_button"
	desc = "unite_atlantia_button_desc"

	visible = { 
		
		any_primary_culture = {
			has_discrimination_trait = amerigan
		}
		
	}

	possible = {

		any_subject_or_below = {

			any_primary_culture = {
				has_discrimination_trait = amerigan
			}

		}
		any_neighbouring_state = {
			owner = {
				any_primary_culture = {
					has_discrimination_trait = amerigan
				}
				OR = {
					market = {
						owner = ROOT
					}
					is_subject_of = ROOT
				}
				owes_obligation_to = ROOT

				
			}
		}
		
		
	}
	effect = {
		every_country = {
			if = {
				limit = {
					is_ai = yes			
					any_primary_culture = {
						has_discrimination_trait = amerigan
					}
					is_subject_of = ROOT
				}	
				annex = PREV
			}
		}
		every_country = {
			if = {
				limit = {
					is_ai = yes			
					any_primary_culture = {
						has_discrimination_trait = amerigan
					}
					OR = {
						owes_obligation_to = ROOT
						market = {
							owner = ROOT
						}
					}
					NOT = {
						is_subject_of = ROOT
					}
					
					this.gdp < root.gdp
				}	
				save_scope_as = atl_subject		
			}
		}
		create_diplomatic_pact = {
			country = scope:atl_subject
			type = protectorate
		}
		remove_diplomatic_pact = {
			country = scope:atl_investment
			type = bankroll
		}
		create_incident = {
			country = ROOT
			value = 6.0
		}
		
		
	}

	ai_chance = {
		base = 100
		 		 
	 }	
}
atlantia_investment_button = {
	name = "atl_invest_button"
	desc = "atl_invest_button_desc"

	visible = { 
		country_has_primary_culture = cu:atlantian
	}

	possible = {
		has_technology_researched = railways	
		has_technology_researched = banking
		NOT = {
			has_modifier = decision_cooldown
		}
		any_neighbouring_state = {
			owner = {
				any_primary_culture = {
					has_discrimination_trait = amerigan
				}
				OR = {
					market = {
						owner = ROOT
					}
					NOT = {
						has_law = law_type:law_isolationism
						has_law = law_type:law_closed_borders
					}
				}
				NOT = {
					has_diplomatic_pact = {
						who = ROOT
						type = bankroll
					}
				}
				
			}
		}
		
	}
	effect = {
		add_modifier = {
			name = decision_cooldown
			months = 60
		}
		every_country = {
			limit = {
				any_primary_culture = {
					has_discrimination_trait = amerigan
				}
				relations:root >= relations_threshold:cordial
				
			}
			create_truce = {
				country = ROOT
				months = 72
			}
			
			save_scope_as = atl_investment
		}
		every_state = {
			limit = {
				state_region = {
					is_homeland = cu:atlantian
				}
			}
			create_building = {
				building = building_railway
			}
		}
		create_diplomatic_pact = {
			country = scope:atl_investment
			type = bankroll
		}
		
		
	}

	ai_chance = {
		base = 100
		 		 
	 }	
}
