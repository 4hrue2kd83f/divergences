assign_state_name_region_pacific_coast = {
	if = {
		limit = {
			state_region = {
				s:STATE_XINGTIAN ?= this
			}
		}
		STATE_XINGTIAN_state_name_assign = yes
	}
	if = {
		limit = {
			state_region = {
				s:STATE_DONGHAI ?= this
			}
		}
		STATE_DONGHAI_state_name_assign = yes
	}
	if = {
		limit = {
			state_region = {
				s:STATE_SANBAO ?= this
			}
		}
		STATE_SANBAO_state_name_assign = yes
	}
}
STATE_XINGTIAN_state_name_assign = { #Colombian
	if = {
		limit = {
			owner = {
				any_primary_culture = {
					cu:colombian ?= this
				}
			}
		}
		set_state_name = STATE_XINGTIAN_colombian
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_colombian
		}
		set_hub_name = {
			type = farm
			name = HUB_NAME_STATE_XINGTIAN_farm_colombian
		}
		set_hub_name = {
			type = wood
			name = HUB_NAME_STATE_XINGTIAN_wood_colombian
		}
		set_hub_name = {
			type = port
			name = HUB_NAME_STATE_XINGTIAN_port_colombian
		}
		set_hub_name = {
			type = mine
			name = HUB_NAME_STATE_XINGTIAN_mine_colombian
		}
	}
	else_if = { #Lusitanian
		limit = {
			owner = {
				any_primary_culture = {
					cu:lusitanian ?= this
				}
				NOT = {
					any_primary_culture = {
						cu:colombian ?= this
					}
				}
			}
		}
		set_state_name = STATE_XINGTIAN_lusitanian
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_lusitanian
		}
		set_hub_name = {
			type = farm
			name = HUB_NAME_STATE_XINGTIAN_farm_lusitanian
		}
		set_hub_name = {
			type = wood
			name = HUB_NAME_STATE_XINGTIAN_wood_lusitanian
		}
		set_hub_name = {
			type = port
			name = HUB_NAME_STATE_XINGTIAN_port_lusitanian
		}
		set_hub_name = {
			type = mine
			name = HUB_NAME_STATE_XINGTIAN_mine_lusitanian
		}
	}
	else_if = { #Cadien
		limit = {
			owner = {
				any_primary_culture = {
					cu:cadien ?= this
				}
			}
		}
		set_state_name = STATE_XINGTIAN_cadien
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_cadien
		}
		set_hub_name = {
			type = farm
			name = HUB_NAME_STATE_XINGTIAN_farm_cadien
		}
		set_hub_name = {
			type = wood
			name = HUB_NAME_STATE_XINGTIAN_wood_cadien
		}
		set_hub_name = {
			type = port
			name = HUB_NAME_STATE_XINGTIAN_port_cadien
		}
		set_hub_name = {
			type = mine
			name = HUB_NAME_STATE_XINGTIAN_mine_cadien
		}
	}
	else_if = { #Buyeoan
		limit = {
			owner = {
				any_primary_culture = {
					cu:chunggu ?= this
				}
				NOR = {
					c:QXI ?= this
					c:QHA ?= this
					c:QDA ?= this
					c:QYO ?= this
					c:QDJ ?= this
					c:QGQ ?= this
					c:QIN ?= this
				}
			}
		}
		set_state_name = STATE_XINGTIAN_buyeoan
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_buyeoan
		}
		set_hub_name = {
			type = farm
			name = HUB_NAME_STATE_XINGTIAN_farm_buyeoan
		}
		set_hub_name = {
			type = wood
			name = HUB_NAME_STATE_XINGTIAN_wood_buyeoan
		}
		set_hub_name = {
			type = port
			name = HUB_NAME_STATE_XINGTIAN_port_buyeoan
		}
		set_hub_name = {
			type = mine
			name = HUB_NAME_STATE_XINGTIAN_mine_buyeoan
		}
	}
	else_if = { #Kaiganese
		limit = {
			owner = {
				any_primary_culture = {
					cu:kaiganese ?= this
				}
			}
		}
		set_state_name = STATE_XINGTIAN_kaiganese
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_kaiganese
		}
		set_hub_name = {
			type = farm
			name = HUB_NAME_STATE_XINGTIAN_farm_kaiganese
		}
		set_hub_name = {
			type = wood
			name = HUB_NAME_STATE_XINGTIAN_wood_kaiganese
		}
		set_hub_name = {
			type = port
			name = HUB_NAME_STATE_XINGTIAN_port_kaiganese
		}
		set_hub_name = {
			type = mine
			name = HUB_NAME_STATE_XINGTIAN_mine_kaiganese
		}
	}
	else = {
		reset_state_name = yes
		reset_hub_names = yes
	}
}
