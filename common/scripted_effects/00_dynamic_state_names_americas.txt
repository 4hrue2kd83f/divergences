STATE_XINGTIAN_state_name_assign = { #Colombian
	if = {
		limit = {
			owner = {
				any_primary_culture = {
					cu:colombian ?= this
				}
			}
		}
		set_state_name = STATE_XINGTIAN_colombian
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_colombian
		}
		set_hub_name = {
			type = farm
			name = HUB_NAME_STATE_XINGTIAN_farm_colombian
		}
		set_hub_name = {
			type = wood
			name = HUB_NAME_STATE_XINGTIAN_wood_colombian
		}
		set_hub_name = {
			type = port
			name = HUB_NAME_STATE_XINGTIAN_port_colombian
		}
		set_hub_name = {
			type = mine
			name = HUB_NAME_STATE_XINGTIAN_mine_colombian
		}
	}
	else_if = { #Lusitanian
		limit = {
			owner = {
				any_primary_culture = {
					cu:lusitanian ?= this
				}
				NOT = {
					any_primary_culture = {
						cu:colombian ?= this
					}
				}
			}
		}
		set_state_name = STATE_XINGTIAN_lusitanian
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_lusitanian

		}
		set_hub_name = {
			type = mine
			name = HUB_NAME_STATE_XINGTIAN_mine_cadien
		}
	}
	else_if = { #Buyeoan/Korean
		limit = {
			owner = {
				any_primary_culture = {
					has_discrimination_trait = korean_language
				}
				NOR = {
					c:QXI ?= this
