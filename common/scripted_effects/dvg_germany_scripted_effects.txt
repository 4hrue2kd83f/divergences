increase_bav_liberal_anger = {

	custom_tooltip  = je_bavarian_constitution_liberal_anger_increase_info

	change_variable = { name = je_bavarian_constitution_liberal_anger add = 1 }

	if = {
		limit = { var:je_bavarian_constitution_liberal_anger > 5 }
		change_variable = { name = je_bavarian_constitution_liberal_anger set = 5 }
	}

	if = {
		limit = { var:je_bavarian_constitution_liberal_anger < 0 }
		change_variable = { name = je_bavarian_constitution_liberal_anger set = 0 }
	}

	if = {
		limit = { var:je_bavarian_constitution_liberal_anger = 0 }
		je:je_bavarian_constitution = {
			set_bar_progress = { value = 0 name = je_bavarian_constitution_liberal_anger_progress_bar }
		}
	}

	if = {
		limit = { var:je_bavarian_constitution_liberal_anger = 1 }
		je:je_bavarian_constitution = {
			set_bar_progress = { value = 1 name = je_bavarian_constitution_liberal_anger_progress_bar }
		}
	}

	if = {
		limit = { var:je_bavarian_constitution_liberal_anger = 2 }
		je:je_bavarian_constitution = {
			set_bar_progress = { value = 2 name = je_bavarian_constitution_liberal_anger_progress_bar }
		}
	}

	if = {
		limit = { var:je_bavarian_constitution_liberal_anger = 3 }
		je:je_bavarian_constitution = {
			set_bar_progress = { value = 3 name = je_bavarian_constitution_liberal_anger_progress_bar }
		}
	}

	if = {
