dvg_belgium.5 = {
	type = country_event
	placement = ROOT

	title = dvg_belgium.5.t
	desc = dvg_belgium.5.d
	flavor = dvg_belgium.5.f

	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}
	immediate = {
		c:PLA = {
			save_scope_as = scope_PLA
		}
	}


	option = {
		name = dvg_belgium.5.a
		default_option = yes
		add_loyalists = {
			value = 0.15
		}
		if = {
			limit = {
				NOT = {
					has_technology_researched = nationalism
				}
			}
			add_technology_progress = {
				technology = nationalism
				progress = 7000
			}
		}
		add_modifier = {
			name = dvg_modifier_armed_forces_bolstered
			months = 30
			is_decaying = yes
		}
		
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = dvg_belgium.5.b
		add_loyalists = {
			value = 0.15
		}
		if = {
			limit = {
				any_subject_or_below = {
					country_has_primary_culture = cu:septiman
				}
			}
			every_country = {
				limit = {
					country_has_primary_culture = cu:septiman
					is_subject = yes
				}
				change_relations = {
					country = THIS
					value = 20
				}
				add_liberty_desire = 25
				
			}
		}

		
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = dvg_belgium.5.c
		add_loyalists = {
			value = 0.15
		}
		if = {
			limit = {
				exists = c:PLA
			}
			change_relations = {
				country = c:PLA
				value = 25
			}
		}
		
		ai_chance = {
			base = 1
		}
	}
}

		}

	duration = 3

	trigger = {
	}
	immediate = {
		c:PLA = {
			save_scope_as = scope_PLA
		}
		if = {
			limit = {
				exists = overlord
			}
			save_scope_as = overlord_scope
		}
	}


	option = {
		name = dvg_belgium.8.a
		default_option = yes
		add_loyalists = {
			value = 0.15
		}
		
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = dvg_belgium.8.b
		trigger = {
			OR = {
				is_player = yes
				liberty_desire >= 50
			}
		}
		create_diplomatic_play = {
			type = dp_independence
			target_country = scope:overlord_scope
			war = no

			add_war_goal = {
				holder = scope:overlord_scope
				type = annex_country
			}
		}
		add_loyalists = {
			value = 0.15
		}
		
		ai_chance = {
			base = 10
		}
	}
}
