	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = { 


	}
	immediate = {
		if = {
			limit = { exists = s:STATE_HOLLAND.region_state:BRG }
			create_country = { #Netherlands
				tag = NET
				origin = this
				state = s:STATE_HOLLAND.region_state:BRG

				
			}
			c:NET = { save_scope_as = scope_NET }
			if = {
				limit = { exists = s:STATE_GELRE.region_state:BRG }
				s:STATE_GELRE.region_state:BRG = { set_state_owner = c:NET }
			}
			if = {
				limit = { exists = s:STATE_FLANDERS.region_state:BRG }
				s:STATE_FLANDERS.region_state:BRG = { set_state_owner = c:NET }
			}
			if = {
				limit = { exists = s:STATE_FRIESLAND.region_state:BRG }
				s:STATE_FRIESLAND.region_state:BRG = { set_state_owner = c:NET }
			}
			
		}
		
	}
	
	
	option = {
		name = dvg_burgundy.43.a
		default_option = yes

			type = dp_annex_war
			target_country = c:NET

		}

		ai_chance = {
			base = 100
		}
	}
	option = {
		name = dvg_burgundy.43.b

			type = dp_annex_war
			target_country = c:NET

		}
		play_as = c:NET

		ai_chance = {
			base = 100
		}
	}
	
}
