# China Reunification
dvg_china.1 = {
	type = country_event
	placement = ROOT

	title = dvg_china.1.t
	desc = dvg_china.1.d
	flavor = dvg_china.1.f

	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
	}

	option = {
		name = dvg_china.1.a
		default_option = yes
		add_loyalists = {
			value = 0.1
			culture = cu:han
		}
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:han
				is_homeland = cu:hakka
				is_homeland = cu:min
				is_homeland = cu:yue
				}
			}
			add_claim = ROOT
		}

		change_infamy = -1000
		change_infamy = 25
		change_tag = CHI
		set_global_variable = china_reunified
	}
}
