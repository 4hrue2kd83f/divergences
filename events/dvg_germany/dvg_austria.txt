### The Carinthian War of Independence
dvg_austria.41 = {

	event_image = { video = "unspecific_military_parade" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"
	duration = 1

	type = country_event
	placement = ROOT
	
	title = dvg_austria.41.t
	desc = dvg_austria.41.d
	flavor = dvg_austria.41.f

	option = {
		name = dvg_austria.41.a
		default_option = yes

		create_country = {
			tag = SLO
			origin = this
			state = s:STATE_SLOVENIA.region_state:AUS
		}

		create_diplomatic_play = {
				
			
			target_country = c:SLO
			type = dp_rebellion_annex
			
			add_war_goal = {
				holder = c:SLO
				type = independence
				target_country = c:AUS
			}

			 
			
		}

		random_diplomatic_play = {
			limit = {
				initiator = c:AUS
				target = c:SLO
			}
			save_scope_as = dvg_aus_slovene_revolt
		}

		s:STATE_SLOVENIA.region_state:SLO = {
			create_building={
				building="building_barracks"
				add_ownership={
					country={
						country="c:SLO"
						levels=15
					}
				}
				reserves=1
				activate_production_methods={ "pm_no_organization" }
			}
		}

		c:SLO = {
			add_modifier = {
				name = dvg_slovenia_revolt
				days = 500
			}
		}

		if = {
			limit = {
				exists = c:CRO
			}

			c:CRO = {
				trigger_event = {
					id = dvg_austria.42
				}
			}

		}
		
		
		
	}


}
