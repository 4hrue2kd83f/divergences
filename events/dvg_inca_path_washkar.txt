			kill_character = yes
		}

			type = je_washkars_reforms
		}
		add_journal_entry = { 
			type = je_washkars_conquests
		}
		trigger_event = { id = dvg_inca_path_washkar.2 days = 180 } 
		hidden_effect = {
			trigger_event = { id = dvg_inca_reconquest.9 days = 720 }
		}
	}
