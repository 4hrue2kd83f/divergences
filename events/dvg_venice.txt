dvg_ven.2 = { #calls for enfranchisement
	type = country_event
	placement = root
	title = dvg_ven.2.t
	desc = dvg_ven.2.d
	flavor = dvg_ven.2.f

	duration = 3

	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/europenorthamerica_political_extremism"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	trigger = {
		any_interest_group = {
			OR = {
				is_powerful = yes
				is_insurrectionary = yes
			}
			leader = { has_ideology = ideology:ideology_radical }
		}
		has_modifier = dvg_ven_patricians
	}

	immediate = {
		
		set_variable = {
			name = ven_patricians_success_count_var 	#journal entry counter/ticker
			value = 0
		}
		set_variable = {
			name = ven_populists_success_count_var #journal entry counter/ticker
			value = 0
		}
	}

	option = {
		name = dvg_ven.2.a
		default_option = yes

		#set_variable = french_path

		add_journal_entry = {
			type = je_ven_patricians_success
		}

		add_journal_entry = {
			type = je_ven_populists_success
		}

		add_loyalists = {
			value = very_large_radicals
			strata = rich
		}
		add_radicals = {
			value = large_radicals
			strata = poor
		}
		add_radicals = {
			value = medium_radicals
			strata = middle
		}
		hidden = {
			owner = {
				trigger_event = {
					id = dvg_ven.101
					days = 45
				}
			}
		}
	}
}

dvg_ven.101 = { #Chose your character
	type = country_event
	placement = ROOT

	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/capitalists_meeting"

	icon = "gfx/interface/icons/event_icons/event_default.dds"
	trigger = {
		#will happen. triggered by dvg_ven.2 disaster start
	}
	immediate = {
		random_interest_group = {
			limit = {
				is_marginal = no
				NOT = { leader = { has_ideology = ideology:ideology_radical } }
				NOR = {
					is_interest_group_type = ig_landowners
					is_interest_group_type = ig_petty_bourgeoisie
					is_interest_group_type = ig_armed_forces
				}
			}
			save_scope_as = dvg_ven_radical_ig
		}
		create_character = {
			first_name = Giancarlo
			last_name = Iannarella
			is_agitator = yes
			add_character_role = politician
			age = 31
			culture = cu:venetian
			interest_group = ig_intelligentsia
			ideology = ideology_positivist
			traits = {
				demagogue
				erudite
			}
			on_created = {
				set_variable = dvg_ven_positivist_leader_var
			}
		}
		create_character = {
			first_name = Felice
			last_name = Russo
			is_agitator = yes
			add_character_role = politician
			age = 37
			culture = cu:venetian
			interest_group = scope:dvg_ven_radical_ig
			ideology = ideology_radical
			traits = {
			}
			on_created = {
				set_variable = dvg_ven_radical_leader_var			
			}
		}
		random_scope_character = {
			limit = {
				has_variable = dvg_ven_positivist_leader_var
			}
			save_scope_as = dvg_ven_positivist_leader_scope
		}
		random_scope_character = {
			limit = {
				has_variable = dvg_ven_radical_leader_var
			}
			save_scope_as = dvg_ven_radical_leader_scope
		}
	}
	option = {	#GARNER ACADEMIC SUPPORT we want the positivist, make the other an agitator
		name = dvg_ven.101.a
		default_option = yes
		
		scope:dvg_ven_positivist_leader_scope = {
			set_as_interest_group_leader = yes
		}
		add_modifier = {
			name = dvg_ven_enter_positivist_leader_var
			months = normal_modifier_time
		}
		hidden = {
			set_variable = dvg_ven_supported_positivist
		}
	}
	option = {	#THEY HAVE A POINT...   we want the radical, make the other an agitator
		name = dvg_ven.101.b
		default_option = yes
		scope:dvg_ven_radical_leader_scope = {
			set_as_interest_group_leader = yes
		}
		hidden = {
			set_variable = dvg_ven_supported_radicals
		}
	}
	option = {	#EXILE THEM ALL
		name = dvg_ven.101.c
		ig:ig_petty_bourgeoisie = {
			add_modifier = {
				name = dvg_ven_patricians_event_no_outside_help
				months = normal_modifier_time
			}
		}
		ig:ig_landowners = {
			add_modifier = {
				name = dvg_ven_patricians_event_no_outside_help
				months = normal_modifier_time
			}
		}
		scope:dvg_ven_positivist_leader_scope = {
			exile_character = yes
		}
		scope:dvg_ven_radical_leader_scope = {
			exile_character = yes
		}
		
	}
}
