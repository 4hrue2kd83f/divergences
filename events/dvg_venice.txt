	trigger = {
		has_journal_entry = je_springtime_of_the_peoples
		has_modifier = dvg_ven_patricians
		scope:empty_agitator_slots >= 1

	immediate = {
		
		set_variable = {
			name = ven_populists_revolt 	#journal entry counter/ticker
			value = 0
		}
		set_variable = {
			name = ven_patricians_revolt 	#journal entry counter/ticker
			value = 0
		}
	}

	option = {
		name = dvg_ven.2.a
		default_option = yes

		#set_variable = french_path

		add_journal_entry = {
			type = je_ven_ease_demands
		}

		add_journal_entry = {
			type = je_ven_populists_revolt
		}

		add_journal_entry = {
			type = je_ven_patricians_revolt
		}
		
		add_loyalists = {
			value = very_large_radicals
			strata = rich
		}
		add_radicals = {
			value = large_radicals
			strata = poor
		}
		add_radicals = {
			value = medium_radicals
			strata = middle
		}
		owner = {
			trigger_event = {
				id = dvg_ven.101
				days = 21
			}
		}
	}
