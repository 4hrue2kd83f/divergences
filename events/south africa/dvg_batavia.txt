namespace = dvg_batavia

#batavia Primer
dvg_batavia.1 = {
	type = country_event	
	placement = ROOT

	title = dvg_batavia.1.t
	desc = dvg_batavia.1.d
	flavor = dvg_batavia.1.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_springtime_of_nation.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/springtime_of_nation"
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 1

	trigger = {
	}

	immediate = {
	}

	option = {
		name = dvg_batavia.1.a
		custom_tooltip = {
		text = dvg_batavia_formed_tt
		}
		trigger_event = { id = dvg_batavia.2 days = 30 }
        c:root = {
        	if = {
        		limit = {
        			c:BTV = {
        				has_law = law_peasant_levies
        			}
        		}
        		add_journal_entry = {
        			type = je_btv_military
        		}
        	}
        }
        c:root = {
			if = {
				limit = {
					country_has_voting_franchise = no
				}
				add_journal_entry = {
        			type = je_btv_autocracy
        		}
			} 
        }
	}
}
