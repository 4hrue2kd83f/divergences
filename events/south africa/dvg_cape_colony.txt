dvg_cape_colony.1 = {
	type = country_event

		video = "gfx/event_pictures/southamerica_war_civilians.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/war_civilians"
	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	title = dvg_cape_colony.1.t
	desc = dvg_cape_colony.1.d
	flavor = dvg_cape_colony.1.f

	duration = 1

	trigger = {
	}

	immediate = {
		add_modifier = {
            name = dvg_geus_resistance
        }
	}

	option = {
		name = dvg_cape_colony.1.a
		hidden_effect = { 
        	remove_modifier = dvg_geus_resistance
        }
        add_modifier = {
            name = dvg_geus_resistance
        }
	}
}

#Leeuwin infiltrates mosselbai
dvg_cape_colony.7 = { 
	type = country_event
	event_image = {
		video = "gfx/event_pictures/europenorthamerica_capitalists_meeting.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	title = dvg_cape_colony.7.t
	desc = dvg_cape_colony.7.d
	flavor = dvg_cape_colony.7.f

	duration = 1

	trigger = {
	}

	option = { #we don't have the resources 
	name = dvg_cape_colony.7.a
		change_variable = {
			name = geus_agitation_var
			add = -10
		}
		c:LEW = {
			trigger_event = { dvg_leeuwin.8 days = 90 }
		}
	}
	option = { #crack down
	name = dvg_cape_colony.7.b
		change_variable = {
			name = geus_agitation_var
			add = 5
		}
		c:LEW = {
			trigger_event = { dvg_leeuwin.9 days = 90 }
		}
	}

}
