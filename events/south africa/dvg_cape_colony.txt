#Cape Colony Primer
dvg_cape_colony.1 = {
	type = country_event
	event_image = {
		video = "gfx/event_pictures/southamerica_war_civilians.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/war_civilians"
	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	title = dvg_cape_colony.1.t
	desc = dvg_cape_colony.1.d
	flavor = dvg_cape_colony.1.f

	duration = 1

	trigger = {
	}

	immediate = {
		add_modifier = {
            name = dvg_geus_resistance
        }
	}

	option = {
		name = dvg_cape_colony.1.a
		hidden_effect = { 
        	remove_modifier = dvg_geus_resistance
        }
        add_modifier = {
            name = dvg_geus_resistance
        }
	}
}

		}
	}
}
