#Trans-oranje settlers
dvg_opstalia.4 = {
	type = country_event
	placement = ROOT

	title = dvg_opstalia.4.t
	desc = dvg_opstalia.4.d
	flavor = dvg_opstalia.4.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_gold_prospectors"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/africa/construction_colony"
	icon = "gfx/interface/icons/event_icons/event_map.dds"


	duration = 3

	trigger = {
	}

	option = { #demand the land
	name = dvg_opstalia.4.a
	default_option = yes
		c:VRY = {
			trigger_event = {id = dvg_vryland.2 days = 14}
		}
		ai_chance = {
			base = 1
		}
    }
	option = { #take everything 
	name = dvg_opstalia.4.b
		custom_tooltip = {
			text = dvg_opstalia_annex_vryland_tt
			create_diplomatic_play = {
				name = dvg_bastaard_war_tt
			
				target_country = c:VRY
				
				war = yes
				
				type = dp_annex_war
				
				add_war_goal = {
					holder = c:OPS
					type = annex_country
					target_country = c:VRY
				}
			}
		}
    }
	option = { #ignore it
	name = dvg_opstalia.4.c
		c:LEW = {
			trigger_event = {id = dvg_leeuwin.2 days = 90}
		}
		prestige = -5
		interest_group_ig_armed_forces_approval_add = -3
		interest_group_ig_landowners_approval_add = -3
		interest_group_ig_rural_folk_approval_add = -2
		ai_chance = {
			base = 1
		}
	}
}

	event_image = {
		video = "gfx/event_pictures/southamerica_war_civilians.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/war_civilians"
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 1

	trigger = {
		c:OPS = {
			owns_entire_state_region = STATE_NOORDLAND
			owns_entire_state_region = STATE_BOTSWANA
		} 
	}

	option = { #keep it
	name = dvg_opstalia.7.a
		ai_chance = {
			base = 1
		}
    }
    option = { #puppet them
	name = dvg_opstalia.7.b
		s:STATE_NOORDLAND = {
			set_state_owner = VRY
        }
        s:STATE_BOTSWANA = {
			set_state_owner = VRY
        }
        create_diplomatic_pact = {
        country = VRY
        type = puppet
        }
        ai_chance = {
			base = 1
		}
    }
