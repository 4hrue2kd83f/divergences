#opstalia moves on reurveld
dvg_nuweflander.2 = {
	type = country_event
	placement = ROOT

	title = dvg_nuweflander.2.t
	desc = dvg_nuweflander.2.d
	flavor = dvg_nuweflander.2.f

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 1

	trigger = {
		c:OPS = {
			active_diplomatic_play = {
				target = c:SUA
			}
			save_scope_as = reurveld_acquisiton
		}
	}

	option = { #they're not a bad neighbour
		name = dvg_nuweflander.2.a
			scope:reurveld_acquisiton = {
				add_target_backers = { c:NFL }
			} 

		ai_chance = {
				if = {
				limit = { 
					c:OPS = {
						relations:root >= 50
					}
				}
				base = 10
			}
			if = {
				limit = { 
					c:OPS = {
						relations:root >= 30
					}
				}
				base = 5
			}
			if = {
				limit = { 
					c:OPS = {
						relations:root <= 30
					}
				}
				base = 1
			}
		}
	}
	option = { #step in to protect them 
		name = dvg_nuweflander.2.b
		ai_chance = {
				if = {
				limit = { 
					c:OPS = {
						relations:root >= 50
					}
				}
				base = 0
			}
			if = {
				limit = { 
					c:OPS = {
						relations:root >= 0
					}
				}
				base = 5
			}
			if = {
				limit = { 
					c:OPS = {
						relations:root <= 0
					}
				}
				base = 10
			}
		}
	}
}
