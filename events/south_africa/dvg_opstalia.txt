#Trans-oranje settlers
dvg_opstalia.4 = {
	type = country_event
	placement = ROOT

	title = dvg_opstalia.4.t
	desc = dvg_opstalia.4.d
	flavor = dvg_opstalia.4.f

	event_image = {
		video = "gfx/event_pictures/africa_prosperous_farm.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/africa/construction_colony"
	icon = "gfx/interface/icons/event_icons/event_map.dds"


	duration = 3

	trigger = {
	}

	option = { #demand the land
	name = dvg_opstalia.4.a
	default_option = yes
		c:VRY = {
			trigger_event = {id = dvg_vryland.2 days = 14}
		}
		ai_chance = {
			base = 1
		}
    }
	option = { #take everything 
	name = dvg_opstalia.4.b
		custom_tooltip = {
			text = dvg_opstalia_annex_vryland_tt
			create_diplomatic_play = {
				name = bastaard_war
				
				escalation = 0

				target_country = c:VRY
				
				war = no
				
				type = dp_annex_war
				
				add_war_goal = {
					holder = c:OPS
					type = annex_country
					target_country = c:VRY
				}
			}
		}
    }
	option = { #ignore it
	name = dvg_opstalia.4.c
		c:LEW = {
			trigger_event = {id = dvg_leeuwin.2 days = 90}
		}
		country_prestige_add = -5
		interest_group_ig_armed_forces_approval_add = -3
		interest_group_ig_landowners_approval_add = -3
		interest_group_ig_rural_folk_approval_add = -2
		ai_chance = {
			base = 1
		}
	}
}

        }

        country = VRY

		}
